## 내 소스
```python
def self_num(n):
  tmp = n
  sum = tmp # 결과값
  # print("sum :",sum)
  # n의 자리 수
  length = len(str(n)) # 만약 n이면 10의 n-1승

  for i in range(length):
    # 0 일때 length - i 승으로 나눈 몫
    this_chip = 10 ** (length-i-1)
    # print("this_chip :",this_chip)
    sum += tmp // this_chip
    # print("after sum:",sum)
    tmp = tmp % this_chip

  return sum

not_self = [self_num(i) for i in range(9969)]

for i in range(9994):
  if(i in not_self):
    continue
  print(i)
```
